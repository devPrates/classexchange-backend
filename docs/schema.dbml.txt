Table Campus {
  id uuid [pk]
  nome varchar [not null]
  sigla varchar(10) [not null]
  email varchar [not null, unique]
  slug varchar(100) [not null, unique]
  telefone varchar
  endereco varchar
}

Table Usuario {
  id uuid [pk]
  nome varchar [not null]
  email varchar [not null, unique]
  senha varchar [not null]
  celular varchar [not null]
  role RoleUsuario [not null]
  campus_id uuid [not null, ref: > Campus.id]
}

Enum RoleUsuario {
  PROFESSOR
  COORDENACAO
  DIRETORENSINO
  COORDENADORCURSO
  ADMINISTRADOR
}

Table Curso {
  id uuid [pk]
  nome varchar [not null]
  sigla varchar(10) [not null]
  slug varchar(100) [not null, unique]
  campus_id uuid [not null, ref: > Campus.id]
}

Table Turma {
  id uuid [pk]
  nome varchar [not null]
  slug varchar(100) [not null, unique]
  numero int [not null]
  curso_id uuid [not null, ref: > Curso.id]
}

Enum TipoPeriodo {
  ANUAL
  SEMESTRAL
  BIMESTRAL
}

Table Periodo {
  id uuid [pk]
  nome varchar [not null]
  slug varchar(100) [not null, unique]
  tipo_periodo TipoPeriodo [not null]
  numero int [not null]
  ano int [not null]
  inicio date [not null]
  fim date [not null]
  turma_id uuid [not null, ref: > Turma.id]
  curso_id uuid [not null, ref: > Curso.id]
}

Table Disciplina {
  id uuid [pk]
  nome varchar [not null]
  slug varchar(100) [not null, unique]
  carga_horaria double
  ementa text
  curso_id uuid [not null, ref: > Curso.id]
  periodo_id uuid [not null, ref: > Periodo.id]
}

Table Local {
  id uuid [pk]
  nome varchar [not null]
}

Enum MedidaTempo {
  MINUTOS
  HORAS
}

Table CargaHoraria {
  id uuid [pk]
  nome varchar [not null]
  duracao int [not null]
  medida_tempo MedidaTempo [not null]
}

Table DisciplinaTurma {
  id uuid [pk]
  disciplina_id uuid [not null, ref: > Disciplina.id]
  turma_id uuid [not null, ref: > Turma.id]
  local_id uuid [not null, ref: > Local.id]
}

Table Horario {
  id uuid [pk]
  dia_da_semana int [not null]
  hora_inicio varchar [not null]
  hora_fim varchar [not null]
  disciplina_turma_id uuid [not null, ref: > DisciplinaTurma.id]
  carga_horaria_id uuid [not null, ref: > CargaHoraria.id]
}

Table Estudante {
  id uuid [pk]
  nome varchar [not null]
  email varchar [not null, unique]
}

Enum SituacaoClasse {
  EM_CURSO
  EVADIDO
}

Table EstudanteCurso {
  id uuid [pk]
  matricula varchar [not null]
  vinculo_curso date [not null]
  situacao SituacaoClasse [not null]
  estudante_id uuid [not null, ref: > Estudante.id]
  curso_id uuid [not null, ref: > Curso.id]
}

Table EstudanteDisciplina {
  id uuid [pk]
  matricula varchar [not null]
  vinculo_curso date [not null]
  situacao SituacaoClasse [not null]
  estudante_id uuid [not null, ref: > Estudante.id]
  disciplina_turma_id uuid [not null, ref: > DisciplinaTurma.id]
}

Table ProfessorCurso {
  id uuid [pk]
  usuario_id uuid [not null, ref: > Usuario.id]
  curso_id uuid [not null, ref: > Curso.id]
}

Table ProfessorDisciplina {
  id uuid [pk]
  inicio date [not null]
  fim date
  usuario_id uuid [not null, ref: > Usuario.id]
  disciplina_turma_id uuid [not null, ref: > DisciplinaTurma.id]

Table CoordenadorCurso {
  id uuid [pk]
  inicio date [not null]
  fim date
  usuario_id uuid [not null, ref: > Usuario.id]
  curso_id uuid [not null, ref: > Curso.id]
}

Table DiretorEnsino {
  id uuid [pk]
  inicio date [not null]
  fim date
  usuario_id uuid [not null, ref: > Usuario.id]
  campus_id uuid [not null, ref: > Campus.id]
